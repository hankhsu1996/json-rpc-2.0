41std::optional<Json> Dispatcher::DispatchSingleRequestInner(
42 const Json &requestJson) {
43 auto validationError = ValidateRequest(requestJson);
44 if (validationError.has_value()) {
45 return validationError->ToJson();
46 }
47
48 Request request = Request::FromJson(requestJson);
49 spdlog::info("Dispatching request: method={}", request.GetMethod());
50
51 auto optionalHandler = FindHandler(handlers_, request.GetMethod());
52 if (!optionalHandler.has_value()) {
53 if (request.GetId().has_value()) {
54 return Response::CreateLibError(
55 LibErrorKind::MethodNotFound, request.GetId())
56 .ToJson();
57 } else {
58 spdlog::warn("Method {} not found for notification", request.GetMethod());
59 return std::nullopt;
60 }
61 }
62
63 return HandleRequest(request, optionalHandler.value());
64}

81std::vector<Json> Dispatcher::DispatchBatchRequestInner(
82 const Json &requestJson) {
83 std::vector<std::future<std::optional<Json>>> futures;
84
85 for (const auto &element : requestJson) {
86 if (enableMultithreading_) {
87 futures.emplace_back(
88 thread_pool_.submit_task([this, element]() -> std::optional<Json> {
89 return DispatchSingleRequestInner(element);
90 }));
91 } else {
92 futures.emplace_back(std::async(
93 std::launch::deferred, [this, element]() -> std::optional<Json> {
94 return DispatchSingleRequestInner(element);
95 }));
96 }
97 }
98
99 std::vector<Json> responses;
100 for (auto &future : futures) {
101 std::optional<Json> response = future.get();
102 if (response.has_value()) {
103 responses.push_back(response.value());
104 }
105 }
106
107 return responses;
108}

110std::optional<Response> Dispatcher::ValidateRequest(const Json &requestJson) {
111 if (!requestJson.is_object()) {
112 return Response::CreateLibError(LibErrorKind::InvalidRequest);
113 }
114
115 if (!requestJson.contains("jsonrpc") || requestJson["jsonrpc"] != "2.0") {
116 return Response::CreateLibError(LibErrorKind::InvalidRequest);
117 }
118
119 if (!requestJson.contains("method")) {
120 if (requestJson.contains("id")) {
121 // Method call without method field, will return an error
122 return Response::CreateLibError(LibErrorKind::InvalidRequest);
123 } else {
124 // Notification without method field, will be ignored
125 spdlog::warn("Request missing method field and id");
126 return std::nullopt;
127 }
128 } else if (!requestJson["method"].is_string()) {
129 return Response::CreateLibError(LibErrorKind::InvalidRequest);
130 }
131
132 return std::nullopt; // No errors found
133}

146std::optional<Json> Dispatcher::HandleRequest(
147 const Request &request, const Handler &handler) {
148 if (request.GetId().has_value()) {
149 // If the request has an ID, it is a method call
150 if (std::holds_alternative<MethodCallHandler>(handler)) {
151 const MethodCallHandler &methodCallHandler =
152 std::get<MethodCallHandler>(handler);
153 Response response = HandleMethodCall(request, methodCallHandler);
154 return response.ToJson();
155 } else {
156 return Response::CreateLibError(
157 LibErrorKind::InvalidRequest, request.GetId())
158 .ToJson();
159 }
160 } else {
161 // Otherwise, it is a notification
162 if (std::holds_alternative<NotificationHandler>(handler)) {
163 const NotificationHandler &notificationHandler =
164 std::get<NotificationHandler>(handler);
165 HandleNotification(request, notificationHandler);
166 return std::nullopt;
167 } else {
168 // Invalid request type for a notification
169 return std::nullopt;
170 }
171 }
172}

184 return Response::CreateLibError(
