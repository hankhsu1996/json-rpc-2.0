95void Client::HandleResponse(const std::string &response) {
96 nlohmann::json jsonResponse;
97 try {
98 jsonResponse = nlohmann::json::parse(response);
99 } catch (const std::exception &e) {
100 spdlog::error("Failed to parse JSON response: {}", e.what());
101 throw std::runtime_error(
102 "Failed to parse JSON response: " + std::string(e.what()));
103 }
104 if (ValidateResponse(jsonResponse)) {
105 int requestId = jsonResponse["id"].get<int>();
106
107 std::lock_guard<std::mutex> lock(pendingRequestsMutex_);
108 auto it = pendingRequests_.find(requestId);
109 if (it != pendingRequests_.end()) {
110 it->second.set_value(jsonResponse);
111 pendingRequests_.erase(it);
112 } else {
113 spdlog::error("Received response for unknown request ID: {}", requestId);
114 throw std::runtime_error("Received response for unknown request ID: " +
115 std::to_string(requestId));
116 }
117 } else {
118 spdlog::error("Invalid JSON-RPC response: {}", jsonResponse.dump());
119 throw std::runtime_error(
120 "Invalid JSON-RPC response: " + jsonResponse.dump());
121 }
122}

124bool Client::ValidateResponse(const nlohmann::json &response) {
125 if (!response.contains("jsonrpc") || response["jsonrpc"] != "2.0") {
126 return false;
127 }
128
129 if (!response.contains("id")) {
130 return false;
131 }
132
133 bool hasResult = response.contains("result");
134 bool hasError = response.contains("error");
135
136 if (hasResult == hasError) { // Mutually exclusive
137 return false;
138 }
139
140 if (hasError) {
141 const nlohmann::json &error = response["error"];
142 if (!error.contains("code") || !error["code"].is_number() ||
143 !error.contains("message") || !error["message"].is_string()) {
144 return false;
145 }
146 }
147
148 return true;
149}
